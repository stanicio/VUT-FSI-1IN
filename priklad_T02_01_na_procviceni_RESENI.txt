clear all
A1=1; f1=1; phi1=0;
%
A2=2;
f2=2;
phi2=-pi/2;


%casova osa t= 100 hodnot od 0 do 1
%t=0:0.01:1 %?????? 101 SPATNE
t=linspace(0,1)


y1=A1*sin(2*pi*f1*t+phi1);
y2=A2*cos(2*pi*f2*t+phi2);    %cosinus


%vykreslit graf a obe funkce do jednoho grafu
plot(t,y1,t,y2)
%zapnout mrizku
grid on
%nazev grafu
title('nazev grafu')
%popis osy x a y
xlabel('t[s]')
ylabel('sin(x), cos(t)')
